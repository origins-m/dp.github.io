sys.argv[] 用法: 是从一个程序外部来获取参数，0指的是代码本身的路径，外部获取多个参数，获得的是一个list（列表），所以可以使用[] 来取元素，其 0 是程序本身，其他参数都是外部传来的参数。(注： sys.argv[] 指令是接受外部参数，无法使用编辑器运行。只能在 dos 命令行里传参。)

argparse : 是命令行解析模块，配合sys.argv[] 使用。

def LanAll(startip):
index = startip.rfind(‘.’) # 找出最右边 . 的索引
ipfirstpart = startip[0:index + 1] # ip 地址中前三位，如 192.168.0.1
intstart = int(startip[index + 1]) # ip 地址最后一位，转为int
f = range(intstart,255)
global g_mutex
g_mutex = threading.Lock() # 初始化互斥锁
for iplastpart in f:
targetip = ipfirstpart + str(iplastpart) # 拼接ip
# 创建线程对象，存为 thread 线程要执行的函数由 target 指定，args 指定参数，
# 可以是元组~。线程号从 1 开始
thread = threading.Thread(target=Lanlp2Name,args=(targetip,))
thread_pool.append(thread)
thread.start()
# 阻塞主线程 collect all threads
for pos in f:
threading.Thread.join(thread_pool[pos - intstart])
#输出结果
hosts = range(0,len(list_of_name))
for host in hosts:
print(list_of_ip[host],’======>’,list_of_name[host])
print(‘Find’,len(list_of_name),’hosts.Done!’)
def Lanlp2Name(ip):
try:
(name, aliaslist,addresslist) = socket.gethostbyaddr(ip)

global g_mutex # 再次声明

list_of_ip.append(ip)
g_mutex.release() # 释放锁
